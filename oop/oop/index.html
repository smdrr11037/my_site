    Reference

    • type& refname = name;
    char c;
char& r = c;//a reference to c, 此时必须引用，can't be null
int& r = 10;//illegal, 只能引用变量
r = 'x';//c也被赋值'x'

    常用于传参（实参）

    void f(int& x);
f(y);//x引用y，函数中x改变，外面y也改变
f(y*3);//illegal
int&* p;//illegal，指向引用变量的指针不合法
void f(int*& p);//引用指针

    Const 常量

    用来代替宏

    const int x = 123;//必须初始化，x保持常数值
x = 27;//illegal
x++;//illegal
const int y = x;//可以任意赋给别人

    编译器会避免给它分配内存：直接写在表达式中

    没法在编译时刻确定\外部要使用时，分配内存

    const string* p1 = "ABC";//指针指向的内容不能更改
//相当于string const* p1
*p1 = "c";//illegal
string *const p2 = "ABC";//指针本身不能更改
p2 = "hello";//OK

    const数组——会分配一片内存，此时编译器不知道数组里的值

    const int i[] = { 1, 2, 3, 4 };
float f[i[3]]; // Illegal  

struct S { int i, j; };
const S s[] = { { 1, 2 }, { 3, 4 } };
double d[s[1].j]; // Illegal

    当需要传递大规模数据（如数组）时，常用指针，容易把原本的数据改掉

    为避免此种情况，会在指针前加const

    //尽量用指针/引用+const
void f(const int*x);//传递的参数不可修改
//函数f中
  x++;//illegal

    自己设计对象

    Point

    //结构体的写法
typedef struct point{
  float x;
  float y;
} Point;

Point a;
a.x = 1;a.y = 2;

print(&a);

void print(const Point* p){
  printf("%d %d\n",p->x,p->y);
}
void move(Point* p,int dx, int dy){
  p->x+=dx;
  p->y+=dy;
}
    • Objects = Attributes + Service
    //类的写法
class Point{
public:
  void init(int x,int y);
  void move(int dx,int dy);
  void print() const;
  //行为和对象封装在一起
private://不写public就是private
  int x;//不允许外界改
  int y;
};

//implementations
void Point::init(int ix,int iy){
  x = ix;
  y = iy;
}
void Point::move(int ix,int iy){
  x+ = dx;
  y+ = dy;
}
void Point::print() const{
  cout << x << ' ' << y << endl;
}
